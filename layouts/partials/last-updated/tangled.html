{{/* Tangled Last Updated Partial Fetches the latest date from a tangled.org Atom feed. Parameters: - appId: App identifier (used for warning messages) - repoUrl: Tangled repository URL (e.g., https://tangled.org/repo) Returns: YYYY-MM-DD formatted date string, or empty string if unavailable */}} {{ $appId := .appId }} {{ $repoUrl := .repoUrl }} {{ $date := "" }} {{/* Build Atom feed URL */}} {{ $atomUrl := printf "%s/feed.atom" (strings.TrimSuffix "/" $repoUrl) }} {{ $result := try (resources.GetRemote $atomUrl) }} {{ with $result.Err }} {{ warnf "[last-updated] app=%s source=tangled: failed to fetch Atom feed from %s: %s" $appId $atomUrl . }} {{ end }} {{ with $result.Value }} {{ $content := .Content }} {{/* Extract the first tag from the Atom feed */}} {{ $match := findRE `([^<]+)` $content 1 }} {{ if $match }} {{ $updated := index $match 0 }} {{ $updated = replaceRE `` "" $updated }} {{ $date = time.Format "2006-01-02" (time $updated) }} {{ end }} {{ end }} {{ return $date }}
