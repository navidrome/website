$tagsData.Value }} {{ $tags := . }} {{ if and $tags (reflect.IsSlice $tags) (gt (len $tags) 0) }} {{ $tag := index $tags 0 }} {{/* Codeberg tag API includes commit info with timestamp */}} {{ with owner/repo from URL: %s" $appId $repoUrl }} {{ end }} {{ return $date }}
